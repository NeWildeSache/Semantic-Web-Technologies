

FUSEKI

apache-jena-fuseki-4.10.0/fuseki-server --file=kulturarena-data.ttl /kulturarena



GENERATE

java -jar sparql-generate-2.1.0.jar -q kulturarena-website-konzerte.rqg -o generated-data.ttl -fo TTL -l DEBUG

./robot -vvv \
    merge --input kulturarena-ontology.ttl \
          --input generated-data.ttl \
    query --update kulturarena-wikidata-mitglied.ru \
          --output kulturarena-data.ttl




REASONING

./robot -vvv \
 merge --input kulturarena-ontology.ttl \
 reason --reasoner hermit \
 --axiom-generators "SubClass EquivalentClass DisjointClasses ClassAssertion" \
 --output kulturarena-inferred.ttl
